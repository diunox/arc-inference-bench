def flatten_dict(d, parent_key='', sep='.'):
    items = []
    for k, v in d.items():
        new_key = f"{parent_key}{sep}{k}" if parent_key else k
        if isinstance(v, dict):
            items.extend(flatten_dict(v, new_key, sep=sep).items())
        else:
            items.append((new_key, v))
    return dict(items)

# Example usage:
nested = {'a': {'b': 1, 'c': {'d': 2}}, 'e': 3}
print(flatten_dict(nested))
# Output: {'a.b': 1, 'a.c.d': 2, 'e': 3}
--------------------------------------------------

This function recursively traverses the dictionary. For each key-value pair, it constructs the new key by appending the current key to the parent key (separated by the given separator, by default a period). If the value is itself a dictionary, it recursively flattens it; otherwise, it adds the key-value pair to the result.
